    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    level=logging.INFO

BOT_TOKEN = "YOUR_BOT_TOKEN_HERE"

TEMP_DIR = "temp_files"
os.makedirs(TEMP_DIR, exist_ok=True)


async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Botni boshlash"""
    await update.message.reply_text(
        "👋 Salom! Men Word fayllarni PDF ga aylantiradigan botman.\n\n"
        "📄 Menga .doc yoki .docx fayl yuboring — men uni PDF ga o'giraman!\n\n"
        "🆓 Xizmat bepul va tez ishlaydi."
    )


async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Yordam"""
    await update.message.reply_text(
        "📋 *Qo'llanma:*\n\n"
        "1️⃣ Menga .doc yoki .docx fayl yuboring\n"
        "2️⃣ Men uni avtomatik PDF ga aylantiraman\n"
        "3️⃣ Tayyor PDF faylni yuklab olasiz\n\n"
        "⚡ O'rtacha vaqt: 5-15 soniya\n"
        "📦 Maksimal fayl hajmi: 20 MB",
        parse_mode="Markdown"
    )


async def convert_document(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Word faylni qabul qilib PDF ga aylantirish"""
    document = update.message.document
    file_name = document.file_name

    # Fayl turini tekshirish
    if not file_name.lower().endswith(('.doc', '.docx')):
        await update.message.reply_text(
            "❌ Faqat .doc yoki .docx fayl yuboring!\n"
            "Boshqa formatlar qo'llab-quvvatlanmaydi."
        )
        return

    # Fayl hajmini tekshirish (20 MB)
    if document.file_size > 20 * 1024 * 1024:
        await update.message.reply_text("❌ Fayl hajmi 20 MB dan katta. Kichikroq fayl yuboring.")
        return

    status_msg = await update.message.reply_text("⏳ Fayl yuklanmoqda...")

    try:
        # Faylni yuklab olish
        file = await context.bot.get_file(document.file_id)
        input_path = os.path.join(TEMP_DIR, file_name)
        await file.download_to_drive(input_path)

        await status_msg.edit_text("🔄 Konvertatsiya qilinmoqda...")

        # LibreOffice orqali PDF ga o'girish
        result = subprocess.run(
            [r"C:\Program Files\LibreOffice\program\soffice.exe", "--headless", ...
             "--outdir", TEMP_DIR, input_path],
            capture_output=True, text=True, timeout=60
        )

        # PDF fayl nomini aniqlash
        pdf_name = os.path.splitext(file_name)[0] + ".pdf"
        pdf_path = os.path.join(TEMP_DIR, pdf_name)

        if result.returncode != 0 or not os.path.exists(pdf_path):
            logger.error(f"LibreOffice xato: {result.stderr}")
            await status_msg.edit_text("❌ Konvertatsiyada xatolik yuz berdi. Iltimos, qayta urinib ko'ring.")
            return

        # PDF ni yuborish
        await status_msg.edit_text("📤 PDF yuborilmoqda...")
        with open(pdf_path, 'rb') as pdf_file:
            await update.message.reply_document(
                document=pdf_file,
                filename=pdf_name,
                caption="✅ Mana PDF faylingiz! Konvertatsiya muvaffaqiyatli yakunlandi."
            )

        await status_msg.delete()

    except subprocess.TimeoutExpired:
        await status_msg.edit_text("❌ Vaqt tugadi. Kichikroq fayl yuboring.")
    except Exception as e:
        logger.error(f"Xatolik: {e}")
        await status_msg.edit_text("❌ Xatolik yuz berdi. Qayta urinib ko'ring.")
    finally:
        # Vaqtinchalik fayllarni o'chirish
        for path in [input_path, pdf_path]:
            if 'path' in dir() and os.path.exists(path):
                try:
                    os.remove(path)
                except:
                    pass


async def unknown_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Noto'g'ri xabar"""
    await update.message.reply_text(
        "📄 Iltimos, menga .doc yoki .docx fayl yuboring.\n"
        "Yordam uchun /help ni bosing."
    )


def main():
    """Botni ishga tushirish"""
    app = ApplicationBuilder().token(BOT_TOKEN).build()

    # Handlerlar
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("help", help_command))
    app.add_handler(MessageHandler(filters.Document.ALL, convert_document))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, unknown_message))

    print("🤖 Bot ishga tushdi!")
    app.run_polling()


if __name__ == "__main__":
    main()
